\begin{practica}

\begin{ejercicio}
  \enunciado Compilar un núcleo monolítico y uno modular con las
  mismas opciones de configuración, comparar los tamaños (tener en
  cuenta el tamaño de los módulos). Sacar conclusiones.

%  \solucion 
\end{ejercicio}

\begin{ejercicio}
  \enunciado Utilizar LILO para hacer múltiples entradas a ambos
  núcleos, además de los núcleos que ya existían en el sistema.

%  \solucion
\end{ejercicio}

\begin{ejercicio}
  \enunciado Probar los núcleos y verificar si existen errores al
  iniciar. Si da errores, ?`qué tipo de errores da?. Si uno de ellos
  únicamente tiene errores y ambos fueron configurados con las mismas
  opciones, ?`por qué cree que da errores?.

%  \solucion
\end{ejercicio}

\begin{ejercicio}
  
  \enunciado Arrancar el sistema con ambos núcleos y comparar los
  tamaños de éstos en memoria. ?`Cuál de ellos tiene ventaja
  en la utilización de recursos?. ?`Y en el tiempo de ejecución?.
  \begin{footnotesize}
    \textbf{Nota}: utilizar \comando{lsmod} para calcular el tamaño
    de los módulos en ejecución y \comando{dmesg | grep Memory} para
    el tamaño del núcleo
  \end{footnotesize}

%  \solucion 
\end{ejercicio}

\end{practica}
